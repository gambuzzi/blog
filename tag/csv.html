utility di console per file csv

Pubblicato: 01 09 2014
Da Roberto Gambuzzi

In Php.

tags: php csv

Traduzioni: en 
#!/usr/bin/env php
<?php

$opt = getopt('f:l:h');
#echo '$opt: ';print_r($opt);
$row = -1;
if (($handle = fopen($opt['f'], "r")) !== FALSE) {
    while (($data = fgetcsv($handle)) !== FALSE) {
        if ($row==-1) $header=$data;
        $row++;
        if ($opt['l']==$row) {
                $num = count($data);
                echo "<p> $num fields in line $row: <br /></p>\n";
                if (array_key_exists('h',$opt)) {
                        print_r(array_combine($header,$data));
                } else {
                        print_r($data);
                }
        }
    }
    fclose($handle);
}
