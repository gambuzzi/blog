$ok=false;
for ($i=1; $i<13; $i+=3) {
    #$diff = (strtotime('now')-strtotime(sprintf('%04i-%02i-01', $y, $i)))/3600/24;
 $diff = (strtotime('now')-strtotime($y."-".$i."-01"))/3600/24;

 }
}
